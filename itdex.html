    </div>

    <div id="result-text"></div>

    <button id="next-btn" onclick="loadNextQuestion()">次の問題へ</button>

<script>

    const quizData = [

        { q: "1979年（誕生年）に発売され、世界中に音楽を持ち歩く文化を作ったのは？", a: ["iPod", "ウォークマン", "ディスクマン", "ラジカセ"], correct: 1 },

        { q: "1983年（4歳）、放課後の過ごし方を一変させた「ファミコン」の正式名称は？", a: ["ファミリーコンピューター", "ホームコンピューター", "ゲームボーイ", "スーパーファミコン"], correct: 0 },

        { q: "1989年（10歳）、ベルリンの壁が崩壊。この時の日本の年号は？", a: ["昭和", "平成", "令和", "大正"], correct: 1 },

        { q: "1993年（14歳）、ドーハの悲劇で出場を逃したスポーツの大会は？", a: ["プロ野球", "ラグビーW杯", "サッカーW杯", "オリンピック"], correct: 2 },

        { q: "1995年（16歳）、女子高生を中心に大流行した足元のアイテムは？", a: ["ハイソックス", "ルーズソックス", "タイツ", "ニーハイ"], correct: 1 },

        { q: "1996年（17歳）、エサをあげないと死んでしまう卵型の携帯ゲームは？", a: ["デジモン", "たまごっち", "ピカチュウ", "テトリス"], correct: 1 },

        { q: "1997年（18歳）、日本中がレオナルド・ディカプリオに熱狂した映画は？", a: ["タイタニック", "アルマゲドン", "スターウォーズ", "アバター"], correct: 0 },

        { q: "1979年に放送開始され、今も続くリアルロボットアニメの金字塔は？", a: ["マジンガーZ", "エヴァンゲリオン", "機動戦士ガンダム", "マクロス"], correct: 2 },

        { q: "中学・高校時代、友達との連絡手段として普及した「数字」でメッセージを送る機械は？", a: ["ポケベル", "ピッチ(PHS)", "スマホ", "糸電話"], correct: 0 },

        { q: "1999年（20歳）、世間を騒がせたコンピューターの誤作動懸念は何問題？", a: ["2000年問題", "ノストラダムス問題", "インターネット問題", "消費税問題"], correct: 0 }

    ];


    let currentIdx = 0;

    let score = 0;


    function showQuestion() {

        const qData = quizData[currentIdx];

        document.getElementById("question-text").innerText = `Q${currentIdx + 1}. ${qData.q}`;

        const optionsArea = document.getElementById("options-area");

        optionsArea.innerHTML = "";

        document.getElementById("result-text").innerText = "";

        document.getElementById("next-btn").style.display = "none";


        qData.a.forEach((ans, i) => {

            const btn = document.createElement("button");

            btn.innerText = ans;

            btn.onclick = () => checkAnswer(i);

            optionsArea.appendChild(btn);

        });

    }


    function checkAnswer(idx) {

        const qData = quizData[currentIdx];

        const resArea = document.getElementById("result-text");

        const btns = document.querySelectorAll("#options-area button");

        btns.forEach(b => b.disabled = true);


        if (idx === qData.correct) {

            resArea.innerText = "⭕ 正解！懐かしいね！";

            resArea.style.color = "#2ecc71";

            score++;

        } else {

            resArea.innerText = "❌ 残念！正解は: " + qData.a[qData.correct];

            resArea.style.color = "#e74c3c";

        }

        document.getElementById("next-btn").style.display = "block";

        if (currentIdx === quizData.length - 1) {

            document.getElementById("next-btn").innerText = "結果発表へ";

        }

    }


    function loadNextQuestion() {

        currentIdx++;

        if (currentIdx < quizData.length) {

            showQuestion();

        } else {

            showFinalResult();

        }

    }


    function showFinalResult() {

        const total = quizData.length;

        const percent = (score / total) * 100;

        let comment = "";


        if (percent >= 80) {

            comment = "完璧！あなたは伝説の79年世代です！🏆合格！";

        } else if (percent >= 40) {

            comment = "うーん、惜しい！<br>本当にあなたは79年生まれ？！🤔";

        } else {

            comment = "記憶が…笑<br>ちゃんとニュースを見ましょう^^";

        }


        document.getElementById("quiz").innerHTML = `

            <h1>結果発表</h1>

            <div class="score-msg">

                あなたの正解数は...

                <span class="score-val">${score} / ${total}</span>

                <p style="font-size:1.2rem; font-weight:bold;">${comment}</p>

            </div>

            <button onclick="location.reload()" style="width:100%; background:#f1c40f; color:#2c3e50; border:none; margin-top:20px; padding:15px; border-radius:10px; cursor:pointer; font-weight:bold; text-align:center;">もう一度挑戦する</button>

        `;

    }


    showQuestion();

</script>
